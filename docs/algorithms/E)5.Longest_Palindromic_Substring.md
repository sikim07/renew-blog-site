## Question.

Given a string `s`, returnÂ *the longest palindromic substring* in `s`.


### Example 1:
```
Input: s = "babad"
Output: "bab"
Note: "aba" is also a valid answer.
```

### Example 2:
```
Input: s = "cbbd"
Output: "bb"
```

### Example 3:
```
Input: s = "a"
Output: "a"
```

### Example 4:
```
Input: s = "ac"
Output: "a"
```

### Constraints: 

- `1 <= s.length <= 1000`
- `s` consist of only digits and English letters.

/**
 * @param {string} s
 * @return {string}
 */
const longestPalindrome = function(s) {
    let odd = findOddChar(s);
    let even = findEvenChar(s);
    return odd.length > even.length ? odd : even;
    
};

const findOddChar = (strings) => {

    let result = strings.charAt(0);
    let current = strings.charAt(0);
    for (let i = 0; i < strings.length; i++) {
        let short = Math.min(strings.length - i, i);        
        if (strings[i-1] === strings[i+1]) {
            for (let j = 1; j <= short; j++) {
                if (strings[i-j] === strings[i+j]) {
                    let newLength = 2 * j + 1;
                    if (current.length < newLength) {
                        current = strings.substr(i-j, 2 * j + 1);
                    }
                } else {
                    break;
                }
            } 
        }
        if (current.length > result.length) {
            result = current;
        }
    }
    return result;
}

const findEvenChar = (strings) => {
    let result = strings.charAt(0);
    let current = strings.charAt(0);
    for (let i = 0; i < strings.length; i++) {
        let short = Math.min(strings.length - i, i);
        if (strings[i] === strings[i+1]) {
           for (let j = 0; j <= short; j++) {
                if (strings[i-j] === strings[i+j+1]) {
                    let newLength = 2 * (j + 1);
                    if (current.length < newLength) {
                        current = strings.substr(i-j, 2 * (j + 1));
                    }
                } else {
                    break    
                };
                if (current.length > result.length) {
                    result = current;
                }
            } 
        }
    }
    return result;
}

public class Solution {
private int lo, maxLen;

public String longestPalindrome(String s) {
	int len = s.length();
	if (len < 2)
		return s;
	
    for (int i = 0; i < len-1; i++) {
     	extendPalindrome(s, i, i);  //assume odd length, try to extend Palindrome as possible
     	extendPalindrome(s, i, i+1); //assume even length.
    }
    return s.substring(lo, lo + maxLen);
}

private void extendPalindrome(String s, int j, int k) {
	while (j >= 0 && k < s.length() && s.charAt(j) == s.charAt(k)) {
		j--;
		k++;
	}
	if (maxLen < k - j - 1) {
		lo = j + 1;
		maxLen = k - j - 1;
	}
}}
